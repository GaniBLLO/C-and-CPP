каждой поступающей строки входного потока, которая бы также удаляла полностью пустые строки*/
#include<stdio.h>
#define MAXLINE 1000

int etline(char line[], int maxline);
void copy(char to[], char from[]);

{

  char line[MAXLINE], longest[MAXLINE];
  max = 0;
  while ((len = etline(line, MAXLINE)) > 0)
    {
      if (len > max){
        max = len;
        copy(longest, line);
      }
    }

    printf("%s%d", longest, max);

}

int etline(char s[], int lim)
  {
    int c,i,chet = 0;
    for(i = 0; i < lim-1 && (c=getchar()) != EOF && c!='\n'; ++i)
      {
        s[i] = c;
      if (c == '\n')
        {
          s[i] = c ;
          ++i;
        }
      else if (c == ' '|| c == '\t')
        {
          ++chet;
          if (chet > 1 && c == '\t')
            {
              putchar( '\b' );
            }
          else 
            {
              s[i] = c;
              ++chet;
              ++i;
            }
        }
      }
    s[i] = '\0';
    return i;
  }

void copy(char to[], char from[])
  {
    int i = 0;
    while ((to[i] = from[i]) != '\0')
    {
      ++i;
    }      
  }
